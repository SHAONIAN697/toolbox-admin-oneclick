set -e
SERVICE="toolbox-admin"
APP="/www/wwwroot/gjx.vst76.cn"
BRANCH="private-local-tested-preserve-data-20260620"
URL="https://raw.githubusercontent.com/SHAONIAN697/toolbox-admin-oneclick/${BRANCH}/packages/toolbox-admin-baota-oneclick-clean-20260622.tar.gz"
SHA="901808ba7eda781c111001afcda3bec0ebd014af98377aef9fa646d101e09367"
TS="$(date +%Y%m%d-%H%M%S)"
PKG="/tmp/toolbox-github-private-local-tested-$TS.tar.gz"
TMP="/tmp/toolbox-github-private-local-tested-$TS"
BACKUP="/www/backup/gjx-toolbox-$TS"

[ -n "$GITHUB_TOKEN" ] || { echo "Set GITHUB_TOKEN first"; exit 1; }
mkdir -p "$TMP" "$BACKUP"
[ -d "$APP" ] || { echo "APP dir not found: $APP"; exit 1; }
curl -L --retry 3 -H "Authorization: Bearer $GITHUB_TOKEN" -o "$PKG" "$URL"
echo "$SHA  $PKG" | sha256sum -c -
tar -xzf "$PKG" -C "$TMP"
python3 -m py_compile "$TMP/ToolboxAdminApi-oneclick/app.py"

cd "$APP"
cp -a app.py wwwroot client-template assets deploy "$BACKUP/" 2>/dev/null || true
rm -rf app.py wwwroot client-template assets deploy __pycache__ data/client-cache data/client-jobs
\cp -a "$TMP/ToolboxAdminApi-oneclick/app.py" "$APP/app.py"
\cp -a "$TMP/ToolboxAdminApi-oneclick/wwwroot" "$APP/wwwroot"
\cp -a "$TMP/ToolboxAdminApi-oneclick/client-template" "$APP/client-template"
\cp -a "$TMP/ToolboxAdminApi-oneclick/assets" "$APP/assets"
\cp -a "$TMP/ToolboxAdminApi-oneclick/deploy" "$APP/deploy"
if [ -d "$BACKUP/wwwroot/uploads" ]; then
  mkdir -p "$APP/wwwroot"
  rm -rf "$APP/wwwroot/uploads"
  \cp -a "$BACKUP/wwwroot/uploads" "$APP/wwwroot/uploads"
fi

systemctl restart "$SERVICE"
sleep 2
systemctl is-active --quiet "$SERVICE"
curl -fsS "http://127.0.0.1:5088/api/public/brand" >/dev/null
grep -q "software_catalog_enabled" app.py
grep -q "ensureButtonsPagePanels" wwwroot/admin.js
grep -q "DefaultMaxParallelDownloads = 5" client-template/ToolboxClient.cs
grep -q "SoftwareCatalogPageId" client-template/ToolboxClient.cs
echo "OK: updated to 20260622 local-tested build from private GitHub branch, data preserved. Refresh admin with Ctrl+F5, then download original/studio/portal EXE again. Backup: $BACKUP"
