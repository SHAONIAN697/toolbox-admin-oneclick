set -e
SERVICE="toolbox-admin"
APP="/www/wwwroot/gjx.vst76.cn"
BRANCH="main"
PKG_NAME="toolbox-admin-baota-oneclick-no-agent-forced32-20260624.tar.gz"
URL_RAW="https://raw.githubusercontent.com/SHAONIAN697/toolbox-admin-oneclick/${BRANCH}/packages/${PKG_NAME}"
URL_GITHUB="https://github.com/SHAONIAN697/toolbox-admin-oneclick/raw/${BRANCH}/packages/${PKG_NAME}"
URL_CODELOAD="https://codeload.github.com/SHAONIAN697/toolbox-admin-oneclick/tar.gz/refs/heads/${BRANCH}"
SHA="ee3fe7732e34fe537e8ee8405febff981b04756c5a1fc36a1441ab433bf41a83"
TS="$(date +%Y%m%d-%H%M%S)"
PKG="/tmp/toolbox-github-private-local-tested-$TS.tar.gz"
TMP="/tmp/toolbox-github-private-local-tested-$TS"
REPO_TMP="/tmp/toolbox-github-private-repo-$TS"
BACKUP="/www/backup/gjx-toolbox-$TS"

[ -n "$GITHUB_TOKEN" ] || { echo "Set GITHUB_TOKEN first"; exit 1; }
mkdir -p "$TMP" "$BACKUP" "$REPO_TMP"
[ -d "$APP" ] || { echo "APP dir not found: $APP"; exit 1; }

download_codeload() {
  local repo_pkg="/tmp/toolbox-github-private-repo-$TS.tar.gz"
  echo "Downloading from private codeload branch archive..."
  rm -f "$repo_pkg"
  rm -rf "$REPO_TMP"
  mkdir -p "$REPO_TMP"
  if ! curl -L --fail --retry 2 --connect-timeout 8 --speed-limit 20480 --speed-time 20 --max-time 180 -H "Authorization: Bearer $GITHUB_TOKEN" -o "$repo_pkg" "$URL_CODELOAD"; then
    return 1
  fi
  if ! tar -xzf "$repo_pkg" -C "$REPO_TMP"; then
    return 1
  fi
  local found
  found="$(find "$REPO_TMP" -path "*/packages/$PKG_NAME" -type f | head -n 1)"
  [ -n "$found" ] || return 1
  cp "$found" "$PKG"
  return 0
}

download_direct() {
  local url="$1"
  echo "Downloading private direct package: $url"
  rm -f "$PKG"
  curl -L --fail --retry 1 --connect-timeout 8 --speed-limit 20480 --speed-time 15 --max-time 60 -H "Authorization: Bearer $GITHUB_TOKEN" -o "$PKG" "$url"
}

download_git() {
  command -v git >/dev/null 2>&1 || return 1
  echo "Downloading through private git clone fallback..."
  rm -rf "$REPO_TMP"
  git clone --depth 1 --branch "$BRANCH" "https://${GITHUB_TOKEN}@github.com/SHAONIAN697/toolbox-admin-oneclick.git" "$REPO_TMP"
  [ -f "$REPO_TMP/packages/$PKG_NAME" ] || return 1
  cp "$REPO_TMP/packages/$PKG_NAME" "$PKG"
}

download_codeload || download_direct "$URL_RAW" || download_direct "$URL_GITHUB" || download_git || { echo "Download failed from all sources"; exit 1; }
echo "$SHA  $PKG" | sha256sum -c -
tar -xzf "$PKG" -C "$TMP"
python3 -m py_compile "$TMP/ToolboxAdminApi-oneclick/app.py"

cd "$APP"
cp -a app.py wwwroot client-template assets deploy admin-desktop-template "$BACKUP/" 2>/dev/null || true
rm -rf app.py wwwroot client-template assets deploy admin-desktop-template __pycache__ data/client-cache data/client-jobs
\cp -a "$TMP/ToolboxAdminApi-oneclick/app.py" "$APP/app.py"
\cp -a "$TMP/ToolboxAdminApi-oneclick/wwwroot" "$APP/wwwroot"
\cp -a "$TMP/ToolboxAdminApi-oneclick/client-template" "$APP/client-template"
\cp -a "$TMP/ToolboxAdminApi-oneclick/assets" "$APP/assets"
\cp -a "$TMP/ToolboxAdminApi-oneclick/deploy" "$APP/deploy"
\cp -a "$TMP/ToolboxAdminApi-oneclick/admin-desktop-template" "$APP/admin-desktop-template"
if [ -d "$BACKUP/wwwroot/uploads" ]; then
  mkdir -p "$APP/wwwroot"
  rm -rf "$APP/wwwroot/uploads"
  \cp -a "$BACKUP/wwwroot/uploads" "$APP/wwwroot/uploads"
fi

systemctl restart "$SERVICE"
sleep 2
systemctl is-active --quiet "$SERVICE"
curl -fsS "http://127.0.0.1:5088/api/public/brand" >/dev/null
grep -q "software_catalog_enabled" app.py
grep -q "ensureButtonsPagePanels" wwwroot/admin.js
grep -q "inviteUseFilter" wwwroot/admin.js
grep -q "/api/admin/popup" app.py
grep -q "savePopupAllBtn" wwwroot/admin.js
! grep -q "popup_sources" app.py
grep -Fq 'public_popup_config(user["id"], self.base_url())' app.py
grep -q "服务器不支持32线程分片" client-template/ToolboxClient.cs
grep -q "return MaxSegmentedDownloadConnections;" client-template/ToolboxClient.cs
! grep -q "DownloadFileSingleConnection(task, attempt);" client-template/ToolboxClient.cs
grep -q "/api/admin/desktop/download" app.py
grep -q "admin-desktop.js" wwwroot/index.html
grep -q "downloadAdminDesktopBtn" wwwroot/index.html
test -f admin-desktop-template/ToolboxAdminDesktop.cs
echo "OK: no-agent forced32 build updated from private GitHub, data preserved. Rebuild/download client EXE. Backup: $BACKUP"
